# 开始使用

安装依赖：

```shell
yarn add sinoui-components @sinouiincubator/tab-group
```

## 入门示例

```tsx
import React from 'react';
import TabGroup, { Tab } from '@sinouiincubator/tab-group';

function Demo() {
  return (
    <TabGroup>
      <Tab title="标签1">标签内容1</Tab>
      <Tab title="标签2">标签内容2</Tab>
      <Tab title="标签3">标签内容3</Tab>
    </TabGroup>
  );
}
```
